原始碼(css_alink.html) 
<html>
<head>
 
  <style>
/* unvisited link */
a:link {
  color: red;
}

/* visited link */
a:visited {
  color: green;
}

/* mouse over link */
a:hover {
  color: hotpink;
}
 
/* selected link */
a:active {
  color: blue;
}
  </style>
</head>
<body>
 <a href="xxx">測試這個連結</a>
</body>
</html>

  h2:hover .xxx,

    visibility: visible;

    color: #00bdf3;
    font-size: 0.5em;
    cursor: pointer;
    visibility: hidden;
    margin-left: 4em;
    position: absolute;
    margin-top: 0.5em;
  }
