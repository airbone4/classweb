最簡單版index_v1.html

外部連結 看內容 
<html>

<head>
	<Title> </Title>
	<script src="https://cdnjs.cloudflare.com/ajax/libs/showdown/1.4.0/showdown.min.js"></script>
	
	<link rel=stylesheet type="text/css" href="css/md_1.css">
	<script language="JavaScript">
		var conv = new showdown.Converter();

		function doMarkdown() {

			var text = markdown.value;

			var htmltext = conv.makeHtml(text)
			html.value = htmltext;
			preview.src = "data:text/html," + htmltext;


		}
	</script>
	
</head>

<body>
	<div class ="raw main">
		<div id="left-column">
			
			<textarea id="markdown" class="full-height" placeholder="Write Markdown"># hello
				</textarea>
		</div>
		<div id="right-column">
			
			<textarea id="html" class="full-height" style="display:none"></textarea>
		<iframe id="preview" class="full-height"></iframe>
		</div>
	</div>

	<button onclick="doMarkdown()">go</button>
	<br>
	
</body>

</html>
